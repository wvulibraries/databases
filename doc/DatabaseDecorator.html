def display_title
  link = title_link
  if popular || trial_database
    "<h3 class='popular-title'> #{link} </h3>".html_safe
  else
    "<h3> #{link} </h3>".html_safe
  end 
end

    "<a href='#{link_to_landing_page}'> #{name} </a>"
  else
    "<a href='#{link_to_database}'> #{name} </a>"
  end 
end
